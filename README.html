public abstract class Duck {
  // HAS-A 叫聲 interface
  QuackBehavior quackBehavior;
  // HAS-A 飛行行為 interface
  FlyBehavior flyBehavior;
  // 外觀
  public abstract void display();
  
  // 執行飛行
  public void performFly() {
    quackBehavior.fly();
  }
  
  // 執行叫
  public void performQuack() {
    quackBehavior.quack();
  }
}
